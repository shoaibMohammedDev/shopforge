   139→   139→function getStatusBadgeVariant(status: string): 'default' | 'secondary' | 'destructive' | 'outline' {
   140→   140→  const map: Record<string, 'default' | 'secondary' | 'destructive' | 'outline'> = {
   141→   141→    PENDING: 'secondary',
   142→   142→    PAID: 'default',
   143→   143→    PROCESSING: 'default',
   144→   144→    SHIPPED: 'default',
   145→   145→    DELIVERED: 'default',
   146→   146→    CANCELLED: 'destructive',
   147→   147→    REFUNDED: 'destructive',
   148→   148→    ACTIVE: 'default',
   149→   149→    INACTIVE: 'secondary',
   150→   150→    DRAFT: 'outline',
   151→   151→    Approved: 'default',
   152→   152→    Rejected: 'destructive',
   153→   153→    Pending: 'secondary',
   154→   154→  }
   155→   155→  return map[status] || 'outline'

   159→   159→  const colorMap: Record<string, string> = {

   167→   167→    ACTIVE: 'bg-green-100 text-green-800 hover:bg-green-100',

   169→   169→    Approved: 'bg-green-100 text-green-800 hover:bg-green-100',
